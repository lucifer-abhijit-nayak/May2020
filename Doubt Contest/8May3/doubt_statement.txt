#include <bits/stdc++.h>

using namespace std;
int main()
{
    int A[3],B[3];
    int cnt=0,cnt1=0;
    for(int i=0;i<3;i++)
       cin>>A[i];
    for(int i=0;i<3;i++)
       cin>>B[i];
    for(int i=0;i<3;i++)
       {
           if(A[i]>B[i])
             cnt++;
            else if(A[i]<B[i])
               cnt1++;
       }
    cout<<cnt<<" "<<cnt1;
}
