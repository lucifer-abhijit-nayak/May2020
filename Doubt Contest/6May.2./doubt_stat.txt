#include<math.h>
#include <bits/stdc++.h>
using namespace std;
int main()
{
    int i,j,c,r;
    signed int sum1=0,sum2=0;
    cin>>c;
    r=c;
    signed int arr[r][c];
    for(i=0;i<r;i++)
    {
         for(j=0;j<c;j++)
         {
            cin>>arr[i][j];
         }
    }
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            if(i==j)
              sum1+=arr[i][j];
            if((i+j)==(r-1))
            {
                sum2+=arr[i][j];
            }
        }
    }
    cout<<abs(sum1-sum2);
    
}
